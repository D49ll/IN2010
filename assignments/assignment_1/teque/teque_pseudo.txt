
proc push_back(x):
    i <- |A| + 1
    A[i] <- x
O(1)

proc push_front(x):
    i <- |A|
    while i > 0 do:
        A[i] <- A[i - 1]
        i <- i - 1
    A[i] <- x
O(n)    

proc push_middle(x):
    j <- |A|
    i <- (|A| + 1 )/ 2

    while j > i do:
        A[j] <- A[j - 1]
        j <- j - 1
    A[j] <- x
O(n/2) -> O(n)    

proc get(i):
    print(A[i])
O(1)
