Pseudo code for kitten assignment

Input: A general tree.
Integers of parent and children nodes.
First line indicates at what node cat is stuck.
The following lines indicates a parent node with its children: 
    a, b_0, b_1 ... b_n
Tree description is finished by a -1 in the last line of input

Output: The path from kittens original position to the tree root.

proc find_index(input):
    cat_position <- input[0]
    path[0] <- cat_position 
    
    i <- 1
    shared_index <- 0

    while input[i] != -1 do:
        parents[shared_index] <- input[i][0]
                
        for j <- 0 to |input[i]| - 1 do
            children[shared_index][j] <- input[i][j + 1]

        i <- i + 1
        shared_index <- shared_index + 1

    shared_index <- 0
    k <- 1

    while shared_index < |children|-1 do:
        for i <-0 to |children[shared_index]| - 1:
            if cat_position = children[shared_index][i]:
                cat_position <- parent[shared_index]
                path[k] <- cat_position
                shared_index <- 0
                k <- k + 1
            else
                shared_index <- shared_index + 1

    return path
